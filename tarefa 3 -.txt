programa resultado_notas
var
	n1, n2, media: real;
início
	escreva ("Informe a primeira nota: ");
	leia (n1);
	
	escreva ("Informe a segunda nota: ");
	leia (n2);

	media = (n1 + n2) / 2;

	se (media >= 7) então
		escreva ("Aprovado");
	senão se (media >= 5) então
		escreva ("Em recuperação");
	senão (media < 5) então
		escreva ("Reprovado");
	fim_se
fim
